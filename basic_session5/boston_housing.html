def performance_metric2(y_true, y_predict):
    """"""

def fit_model(X, y):
    """  [X,y]"""
    cross_validator = KFold(10) # 10
    
    regressor = DecisionTreeRegressor()

    params = {'max_depth':range(1,11)}

def splitData(df, k):
    """ k """
    n = len(df)
    if n < k: #
        raise Exception
        
    df_list = []
    num = n/k
    mod = n%k
    # k
    for i in range(k):
        start = num*i
        end = num*(i+1)
        df_list.append(df[start:end])
        
    #     
    if mod != 0:
        last_df = df_list.pop()
        df_list.append(pd.concat([last_df,df[-mod:]]))

    return df_list
    

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    scores = [0]*cv
    X_list = splitData(X, cv)# cv
    y_list = splitData(y, cv)# cv
    # cv
    for i in range(cv):
        # 
        X_test = X_list.pop()
        X_train = pd.concat(X_list)
        y_test = y_list.pop()
        y_train = pd.concat(y_list)
        # 
        model = estimator.fit(X_train, y_train)
        # 
        y_predict = model.predict(X_test)
        score = performance_metric(y_test, y_predict)
        scores[i] = score
        # 
        X_list.insert(0, X_test)
        y_list.insert(0, y_test)
    
    return scores

def fit_model2(X, y):
    """  [X,y]"""
    
    #
    best_estimator = None
    max_score = 0
    for max_depth in range(1,11): # 110
        regressor = DecisionTreeRegressor(max_depth=max_depth)
        scores = cross_val_score(regressor, X, y)
        mean_score = sum(scores) / len(scores)
        if mean_score > max_score:
            max_score = mean_score
            best_estimator = regressor
    
    return best_estimator
