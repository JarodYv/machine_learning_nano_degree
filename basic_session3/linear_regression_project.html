"""

def gj_Solve(A, b, decPts=4, epsilon = 1.0e-16):
    if len(A) != len(b): # AbNone
        return None
    
    M = augmentMatrix(A, b) # 
    matxRound(M) # 
    (h, w) = shape(M)
    for i in range(0, h):
        '''
        
        '''
        maxrow = i
        for j in range(i+1, h):
            if abs(M[j][i]) > abs(M[maxrow][i]):
                maxrow = j
        swapRows(M, i, maxrow)
        if abs(M[i][i]) < epsilon: # This is a singular matrix
            return None
        
        '''
        
        '''
        for j in range(i+1, h):
            scale = -M[j][i]/M[i][i]
            addScaledRow(M,j,i,scale)

    '''
    
    '''
    for i in range(h-1,-1,-1):
        for j in range(0,i):
            scale = -M[j][i]/M[i][i]
            addScaledRow(M,j,i,scale)
        M[i][w-1] /= M[i][i]
        M[i][i] = 1

    result = [[row[w-1]] for row in M]
    return result
